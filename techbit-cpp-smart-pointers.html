C++ Smart Pointers

Passing Smart Pointers to function

for std::unique_ptr: 
  void by_value(std::unique_ptr < Employee > p) {
    // ...
  }
  void by_ref( std::unique_ptr< Employee > & p ) {
    // ...
    p.reset( new Employee() );
    // ...
  }

  int main(){
    auto p = std::make_unique< Employee >();

    by_value( std::move(p) );    // (1) OK
    by_value( p );               // (2) ERROR - Cannot make Copy

    auto q = std::make_unique< Employee >();
    by_ref( q )                  // OK
    by_ref( std::move ( q ) );   // ERROR - rvalue to non-const lvalue reference NOT OK
  }
for std::shared_ptr: 
  // intention is to part-own;
  void shared_pass_by_value(std::shared_ptr < Employee > p) { // ref count incremented
    // ...
  } // ref count decreased

  // possibly will re-seat the shared-pointer
  void shared_by_ref( std::shared_ptr< Employee > & p ) {
    // ...
    p.reset( new Employee() );
    // ...
  }

  // only alias; so ref count not incremented
  void shared_by_const_ref( const std::shared_ptr< Employee > & p ) {
    // ...
  }
